Change the view based on which item is selected. If no item is selected, then clear and hide the view.

 
    changeSelected: (item) =>

      if item.get('selected')

        @currentItem = item

        this.loadContent item

        $(@el).show()

        this.trigger 'show'

      else
        this.$('.content').empty()
        $(@el).hide()
        this.trigger 'hide'

    close: =>
